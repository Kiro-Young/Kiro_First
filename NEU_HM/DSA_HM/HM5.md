void Change(ElemType array[], int k, int n){
    ElemType e;
    int i = 0, j, p;
    while (i < n - k){
        p = i / k + 1;
        for (j = 0; j < k; j++){
            e = array[j];
            array[j] = array[(p * k + j) % n];
            array[(p * k + j) % n] = e;
            i++;
        }
    }
}
